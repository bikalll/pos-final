rules_version = '2';

// Firebase Storage Security Rules
// These rules allow authenticated users to upload and access images
service firebase.storage {
  match /b/{bucket}/o {
    // Allow authenticated users to upload images to restaurant folders
    match /restaurants/{restaurantId}/{allPaths=**} {
      allow read, write: if request.auth != null && 
        request.auth.token.restaurantId == restaurantId;
    }
    
    // Allow users to upload their own profile images
    match /users/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && 
        request.auth.uid == userId;
    }
    
    // Allow authenticated users to upload general images
    match /uploads/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to upload attendance photos
    match /attendance/{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
